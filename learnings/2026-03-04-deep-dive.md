### 2. 错误处理不足
- 网络超时只返回 null，没有错误信息
- 没有重试机制
- 没有记录失败原因

**改进方案**:
```typescript
interface ScrapeError {
  type: 'network' | 'parse' | 'not_found';
  message: string;
  url: string;
  timestamp: string;
}

async function scrapeWithRetry(url: string, maxRetries = 3): Promise<ScrapedResult | ScrapeError> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      const result = await scrapeFromResultUrl(url);
      if (result) return result;
    } catch (error) {
      if (i === maxRetries - 1) {
        return { type: 'network', message: error.message, url, timestamp: new Date().toISOString() };
      }
      await sleep(1000 * (i + 1)); // 指数退避
    }
  }
}
```

### 3. 没有缓存机制
- 每次抓取都重新请求
- 相同 URL 重复抓取浪费资源

**改进方案**:
```typescript
// 简单的内存缓存
const cache = new Map<string, { data: ScrapedResult; timestamp: number }>();
const CACHE_TTL = 24 * 60 * 60 * 1000; // 24 小时

async function scrapeWithCache(url: string): Promise<ScrapedResult | null> {
  const cached = cache.get(url);
  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.data;
  }
  
  const result = await scrapeFromResultUrl(url);
  if (result) {
    cache.set(url, { data: result, timestamp: Date.now() });
  }
  return result;
}
```
